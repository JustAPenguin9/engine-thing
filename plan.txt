  active: bool
  data: Data
  refreshrate: ms
  start_time: time
  frame_count: uint
  frames: Vec<Scene>

Scene:
  name: String
  done: bool
  handles_events: bool
  transparent: bool
  offset: (x, y)
  frame_count: uint
  draw: fn (&mut App)

animations do not handle events and are transparent unless fullscreen
dialogues do handle events but are transparent
new scenes handle events and are not transparent
